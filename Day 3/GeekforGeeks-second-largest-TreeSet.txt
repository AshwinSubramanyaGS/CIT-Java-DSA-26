import java.util.TreeSet;

class Solution {
    public int getSecondLargest(int[] arr) {

        if (arr.length < 2)
            return -1;
        TreeSet<Integer> set = new TreeSet<>();
        for (int num : arr) {
            set.add(num);
        }
        if (set.size() < 2)
            return -1;
        set.pollLast();      // Remove the largest element
        return set.last();   // Return the new largest (second largest)
    }
}
