import java.util.Arrays;
class Solution {
    public int getSecondLargest(int[] arr) {
        if(arr.length<1) return -1;
        Arrays.sort(arr);
        
        int i=arr.length-2;
        
        while(i>-1&&arr[i]==arr[arr.length-1])i--;
        
        return i>=0? arr[i] : -1;
        
    }
}
